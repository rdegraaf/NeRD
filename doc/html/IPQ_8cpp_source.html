00058 boost::uint16_t ip_checksum(boost::uint32_t init, boost::uint8_t* buf, std::size_t len)
00059 {
00060     boost::uint32_t sum = init;
00061     boost::uint16_t* shorts = reinterpret_cast<boost::uint16_t*>(buf);
00062 
00063     while (len > 1)
00064     {
00065         sum += *shorts++;
00066         len -= 2;
00067     }
00068 
00069     if (len == 1)
00070         sum += *reinterpret_cast<boost::uint8_t*>(shorts);
00071 
00072     while (sum >> 16)   
00073         sum = (sum >> 16) + (sum & 0xFFFF);
00074 
00075     return ~sum;
00076 }
