00217 static int open_devnull(int fd) {
00218   FILE *f = 0;
00219 
00220   if (!fd) f = freopen(_PATH_DEVNULL, "rb", stdin);

00236   int         fd, fds;

00239   /* Make sure all open descriptors other than the standard ones are closed */
00240   if ((fds = getdtablesize()) == -1) fds = OPEN_MAX;
